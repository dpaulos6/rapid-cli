alias ptrf='prettierWriteFunction'
alias format='prettierWriteFunction'
alias pformat='prettierWriteFunction'
alias prettierf='prettierWriteFunction'
alias prettier-format='prettierWriteFunction'

alias ptrw='prettierWriteFunction'
alias write='prettierWriteFunction'
alias pwrite='prettierWriteFunction'
alias prettierw='prettierWriteFunction'
alias prettier-write='prettierWriteFunction'

alias ptrc='prettierCheckFunction'
alias check='prettierCheckFunction'
alias pcheck='prettierCheckFunction'
alias prettierc='prettierCheckFunction'
alias prettier-check='prettierCheckFunction'

prettierWriteFunction() {
  if [ "$1" != "" ]; then
    pnpm exec prettier "$1" --write
  else
    pnpm exec prettier . --write
  fi
}

prettierCheckFunction() {
  if [ "$1" != "" ]; then
    pnpm exec prettier "$1" --check
  else
    pnpm exec prettier . --check
  fi
}
